\PoemTitle{Spiegelsaal}

\begin{verse}

Nicht jede Tür\\
muss geöffnet werden\\
nicht jeder Schlüssel\\
will gefunden sein

O reiner Widerspruch\\
das Fehlen der Achse\\
so vieler Dinge zu sein\\
in niemandes Traum

\end{verse}

\PoemTitle{Hall of Mirrors}
\label{hall-of-mirrors}

\begin{verse}

Not every door\\
needs to be opened\\
not every key\\
wants to be found

O pure contradiction\\
to be the lack of an axis\\
for so many things\\
in nobody's dream

\end{verse}
