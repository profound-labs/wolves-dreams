
\PoemTitle{Sostenuto}

\begin{verse}
A voice, listening\\
voice, sensing us\\
with gentle fingers;

nobody's voice,\\
stray dog\\
seeking a home;

child's voice,\\
little stutter-bug,\\
crawling on the blistered heart:

the squirrel's\\
panicky tracks\\
through our winter-garden;

the dunnock's silent\\
curiosity,\\
its soundless escape.

Who speaks?
\end{verse}
