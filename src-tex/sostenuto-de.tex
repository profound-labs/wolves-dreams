
\PoemTitle{Sostenuto}

\begin{verse}
Eine Stimme,\\
lauschende Stimme,\\
tastet uns ab;

niemandes Stimme,\\
herrenloser Hund\\
auf der Suche nach Heimat;

Kinderstimme,\\
kleiner Stotterkäfer\\
auf der Herzwand des Leidens:

des Eichhörnchens\\
panische Bahnen\\
in unserem Wintergarten;

der Heckenbraunelle\\
lautlose Neugier,\\
lautlose Flucht.

Wer spricht?
\end{verse}
